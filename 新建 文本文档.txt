# 1) 创建独立虚拟环境
python -m venv /workspace/vllm_bench_venv
source /workspace/vllm_bench_venv/bin/activate

# 2) 升级 pip（只在 venv 内）
pip install -U pip

# 3) 安装 vllm（只影响 venv）
pip install vllm

# 4) 验证
vllm --help
vllm bench --help


# 通用最快命令（阿里云源，最稳）
pip install 你要装的包 -i https://mirrors.aliyun.com/pypi/simple/ --trusted-host mirrors.aliyun.com

# 例子：安装 numpy
pip install numpy -i https://mirrors.aliyun.com/pypi/simple/ --trusted-host mirrors.aliyun.com
